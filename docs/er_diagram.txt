Table users {
  id int [pk, increment]
  full_name varchar(100)
  email varchar(150) [unique]
  phone varchar(15)
  password_hash varchar(255)
  role enum('customer', 'admin')
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

Table categories {
  id int [pk, increment]
  name varchar(100)
  slug varchar(120) [unique]
  description text
  image_url varchar(500)
  parent_id int [ref: > categories.id]
  is_active boolean
  created_at timestamp
}

Table products {
  id int [pk, increment]
  title varchar(255)
  description text
  price decimal(10,2)
  discount_price decimal(10,2)
  stock int
  category_id int [ref: > categories.id]
  product_brand varchar(100)
  product_type enum('supplement', 'equipment')
  tax_percentage decimal(5,2)
  gst_applied boolean
  sku varchar(50) [unique]
  weight_grams int
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

Table product_images {
  id int [pk, increment]
  product_id int [ref: > products.id]
  image_url varchar(500)
  alt_text varchar(255)
  is_primary boolean
  sort_order int
}

Table addresses {
  id int [pk, increment]
  user_id int [ref: > users.id]
  label varchar(50)
  recipient_name varchar(100)
  phone varchar(15)
  address_line1 varchar(255)
  address_line2 varchar(255)
  landmark varchar(150)
  city varchar(100)
  state varchar(100)
  pin_code char(6)
  is_default boolean
  created_at timestamp
}

Table coupons {
  id int [pk, increment]
  code varchar(30) [unique]
  description varchar(255)
  discount_type enum('percentage', 'flat')
  discount_value decimal(10,2)
  min_order_value decimal(10,2)
  max_discount decimal(10,2)
  usage_limit int
  times_used int
  valid_from datetime
  valid_until datetime
  is_active boolean
  created_at timestamp
}

Table cart {
  id int [pk, increment]
  user_id int [ref: > users.id]
  product_id int [ref: > products.id]
  quantity int
  added_at timestamp
}

Table wishlist {
  id int [pk, increment]
  user_id int [ref: > users.id]
  product_id int [ref: > products.id]
  added_at timestamp
}

Table orders {
  id int [pk, increment]
  user_id int [ref: > users.id]
  address_id int [ref: > addresses.id]
  coupon_id int [ref: > coupons.id]
  subtotal decimal(10,2)
  tax_amount decimal(10,2)
  discount_amount decimal(10,2)
  shipping_fee decimal(10,2)
  total_price decimal(10,2)
  status enum('Pending', 'Confirmed', 'Processing', 'Shipped', 'Out for Delivery', 'Delivered', 'Cancelled', 'Returned')
  notes text
  created_at timestamp
  updated_at timestamp
}

Table order_items {
  id int [pk, increment]
  order_id int [ref: > orders.id]
  product_id int [ref: > products.id]
  quantity int
  unit_price decimal(10,2)
  tax_percentage decimal(5,2)
  tax_amount decimal(10,2)
  total_price decimal(10,2)
}

Table payments {
  id int [pk, increment]
  order_id int [ref: > orders.id]
  payment_method enum('COD', 'UPI', 'Credit Card', 'Debit Card', 'Net Banking', 'Wallet')
  transaction_id varchar(100)
  upi_id varchar(100)
  amount decimal(10,2)
  status enum('Pending', 'Success', 'Failed', 'Refunded')
  paid_at datetime
  created_at timestamp
}

Table order_tracking {
  id int [pk, increment]
  order_id int [ref: > orders.id]
  status varchar(100)
  location varchar(200)
  description text
  tracked_at timestamp
}

Table reviews {
  id int [pk, increment]
  product_id int [ref: > products.id]
  user_id int [ref: > users.id]
  rating tinyint
  title varchar(200)
  comment text
  is_verified boolean
  created_at timestamp
}

Table admin_logs {
  id int [pk, increment]
  admin_id int [ref: > users.id]
  action varchar(100)
  entity_type varchar(50)
  entity_id int
  details text
  ip_address varchar(45)
  performed_at timestamp
}
